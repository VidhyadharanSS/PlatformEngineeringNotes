### Retry Pattern

```python
import time
import requests

def fetch_with_retry(url, max_retries=3):
    for attempt in range(max_retries):
        try:
            response = requests.get(url, timeout=5)
            response.raise_for_status()
            return response.json()
        except requests.exceptions.RequestException as e:
            if attempt == max_retries - 1:
                raise
            time.sleep(2 ** attempt)  # Exponential backoff
```

### Question 1

**Scenario:** A Python script makes HTTP requests to an external API that occasionally returns 429 (Too Many Requests). The script should retry after waiting the time specified in the `Retry-After` header.

**Question:** Write a function that implements this with proper retry logic.

**Answer:**

```python
import requests
import time
import logging

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

def request_with_retry(url, max_retries=5, **kwargs):
    """
    Make HTTP request with retry on 429 (Rate Limit)
    Uses Retry-After header if available
    """
    for attempt in range(max_retries):
        try:
            response = requests.get(url, **kwargs)
            
            # Success (2xx)
            if response.ok:
                return response
            
            # Rate limited
            if response.status_code == 429:
                retry_after = int(response.headers.get('Retry-After', 60))
                logger.warning(f"Rate limited. Retry after {retry_after}s (attempt {attempt + 1}/{max_retries})")
                time.sleep(retry_after)
                continue
            
            # Other errors (4xx, 5xx)
            response.raise_for_status()
            
        except requests.exceptions.RequestException as e:
            logger.error(f"Request failed (attempt {attempt + 1}): {e}")
            if attempt == max_retries - 1:
                raise
            wait = 2 ** attempt  # Exponential backoff
            time.sleep(wait)
    
    raise Exception(f"Failed after {max_retries} attempts")

# Usage
try:
    response = request_with_retry('https://api.example.com/data', timeout=10)
    data = response.json()
    print(data)
except Exception as e:
    print(f"Failed: {e}")
```

### Question 2

**Scenario:** A script processes a list of URLs. Some URLs might be unreachable, some might time out. The script should:
- Log each URL being processed
- Log errors with appropriate level (WARNING for timeouts, ERROR for other failures)
- Continue processing remaining URLs even if one fails

**Question:** Write the implementation with proper logging.

**Answer:**

```python
import logging
import requests
from requests.exceptions import Timeout, ConnectionError, HTTPError

# Configure logging
logging.basicConfig(
    level=logging.INFO,
    format='%(asctime)s - %(levelname)s - %(message)s'
)
logger = logging.getLogger(__name__)

def check_urls(urls, timeout=10):
    """
    Check multiple URLs and log results.
    Continues processing even if individual URLs fail.
    """
    results = {}
    
    for url in urls:
        logger.info(f"Checking: {url}")
        
        try:
            response = requests.get(url, timeout=timeout)
            response.raise_for_status()
            
            results[url] = {
                'status': 'success',
                'status_code': response.status_code,
                'response_time': response.elapsed.total_seconds()
            }
            logger.info(f"✓ {url} - {response.status_code} ({response.elapsed.total_seconds():.2f}s)")
            
        except Timeout:
            logger.warning(f"Timeout for {url} after {timeout}s")
            results[url] = {'status': 'timeout', 'error': 'Request timed out'}
            
        except ConnectionError:
            logger.error(f"Connection error for {url}")
            results[url] = {'status': 'connection_error', 'error': 'Failed to connect'}
            
        except HTTPError as e:
            logger.error(f"HTTP {e.response.status_code} for {url}")
            results[url] = {
                'status': 'http_error',
                'status_code': e.response.status_code,
                'error': str(e)
            }
            
        except Exception as e:
            logger.exception(f"Unexpected error for {url}")
            results[url] = {'status': 'unknown_error', 'error': str(e)}
    
    # Summary
    successful = sum(1 for r in results.values() if r['status'] == 'success')
    logger.info(f"Summary: {successful}/{len(urls)} successful")
    
    return results

# Usage
urls = [
    'https://google.com',
    'https://nonexistent.example.com',
    'https://httpbin.org/status/404',
    'https://httpbin.org/delay/30'  # Will timeout
]

results = check_urls(urls, timeout=5)
```

### Question 3

**Scenario:** A script uses a third-party API that requires authentication with a Bearer token. The token expires after 1 hour. The script should refresh the token automatically when needed.

**Question:** Design a client class that handles token refresh automatically.

**Answer:**

```python
import requests
import time
import logging
from datetime import datetime, timedelta

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

class APIClient:
    def __init__(self, base_url, client_id, client_secret):
        self.base_url = base_url.rstrip('/')
        self.client_id = client_id
        self.client_secret = client_secret
        self.token = None
        self.token_expires_at = None
        self.session = requests.Session()
    
    def _get_token(self):
        """Obtain new bearer token"""
        logger.info("Obtaining new access token")
        
        response = requests.post(
            f"{self.base_url}/oauth/token",
            json={
                'client_id': self.client_id,
                'client_secret': self.client_secret,
                'grant_type': 'client_credentials'
            }
        )
        response.raise_for_status()
        
        data = response.json()
        self.token = data['access_token']
        expires_in = data.get('expires_in', 3600)
        self.token_expires_at = datetime.now() + timedelta(seconds=expires_in - 60)  # Refresh 1 min early
        
        logger.info(f"Token obtained, expires at {self.token_expires_at}")
    
    def _ensure_token(self):
        """Ensure token is valid, refresh if needed"""
        if not self.token or datetime.now() >= self.token_expires_at:
            self._get_token()
            self.session.headers.update({'Authorization': f'Bearer {self.token}'})
    
    def request(self, method, endpoint, **kwargs):
        """Make authenticated request, auto-refresh token"""
        self._ensure_token()
        
        url = f"{self.base_url}/{endpoint.lstrip('/')}"
        
        try:
            response = self.session.request(method, url, **kwargs)
            
            # If token expired during request (401), retry once
            if response.status_code == 401:
                logger.warning("Token expired, refreshing and retrying")
                self._get_token()
                self.session.headers.update({'Authorization': f'Bearer {self.token}'})
                response = self.session.request(method, url, **kwargs)
            
            response.raise_for_status()
            return response
            
        except requests.exceptions.RequestException as e:
            logger.error(f"API request failed: {e}")
            raise
    
    def get(self, endpoint, **kwargs):
        return self.request('GET', endpoint, **kwargs)
    
    def post(self, endpoint, **kwargs):
        return self.request('POST', endpoint, **kwargs)

# Usage
client = APIClient(
    base_url='https://api.example.com',
    client_id='your_client_id',
    client_secret='your_client_secret'
)

# First request (obtains token)
response = client.get('/users')
print(response.json())

# Second request (uses cached token)
response = client.post('/users', json={'name': 'Alice'})

# Token auto-refreshes when expired
```

### Question 4

**Scenario:** A script calls multiple APIs and needs to log structured data (JSON) for later analysis in a log aggregation system like ELK.

**Question:** Implement JSON logging with extra fields (request_id, user_id, duration).

**Answer:**

```python
import logging
import json
import time
import uuid
from datetime import datetime

class JSONFormatter(logging.Formatter):
    """Custom formatter that outputs JSON"""
    
    def format(self, record):
        log_entry = {
            'timestamp': datetime.utcnow().isoformat(),
            'level': record.levelname,
            'logger': record.name,
            'message': record.getMessage(),
            'module': record.module,
            'line': record.lineno
        }
        
        # Add extra fields from 'extra' dictionary
        if hasattr(record, 'request_id'):
            log_entry['request_id'] = record.request_id
        if hasattr(record, 'user_id'):
            log_entry['user_id'] = record.user_id
        if hasattr(record, 'duration_ms'):
            log_entry['duration_ms'] = record.duration_ms
        
        # Add exception info if present
        if record.exc_info:
            log_entry['exception'] = self.formatException(record.exc_info)
        
        return json.dumps(log_entry)

def setup_json_logging():
    """Configure JSON logging"""
    handler = logging.StreamHandler()
    handler.setFormatter(JSONFormatter())
    
    logger = logging.getLogger()
    logger.addHandler(handler)
    logger.setLevel(logging.INFO)
    
    # Remove default handlers
    for h in logger.handlers[:]:
        if isinstance(h, logging.StreamHandler) and h != handler:
            logger.removeHandler(h)

def api_call_logger(func):
    """Decorator to log API calls with timing"""
    def wrapper(*args, **kwargs):
        request_id = str(uuid.uuid4())[:8]
        start = time.time()
        
        # Create logger with extra context
        logger = logging.getLogger('api')
        logger = logging.LoggerAdapter(logger, {'request_id': request_id})
        
        logger.info(f"Starting API call: {func.__name__}")
        
        try:
            result = func(*args, **kwargs)
            duration_ms = (time.time() - start) * 1000
            
            # Log success with timing
            extra = {'duration_ms': duration_ms, 'request_id': request_id}
            logger.info(f"API call completed: {func.__name__}", extra=extra)
            return result
            
        except Exception as e:
            duration_ms = (time.time() - start) * 1000
            logger.error(f"API call failed: {func.__name__} - {e}", 
                        extra={'duration_ms': duration_ms, 'request_id': request_id})
            raise
    
    return wrapper

# Setup
setup_json_logging()
logger = logging.getLogger('myapp')

# Example usage
@api_call_logger
def fetch_user_data(user_id):
    logger.info(f"Fetching user {user_id}", extra={'user_id': user_id})
    # Simulate API call
    time.sleep(0.5)
    return {'id': user_id, 'name': 'Alice'}

# Call the function
user = fetch_user_data(123)

# Sample JSON output:
# {"timestamp": "2024-01-15T10:30:45.123456", "level": "INFO", "logger": "api", "message": "Starting API call: fetch_user_data", "request_id": "abc12345", "module": "script", "line": 45}
# {"timestamp": "2024-01-15T10:30:45.654321", "level": "INFO", "logger": "myapp", "message": "Fetching user 123", "user_id": 123, "request_id": "abc12345", ...}
# {"timestamp": "2024-01-15T10:30:45.654321", "level": "INFO", "logger": "api", "message": "API call completed: fetch_user_data", "duration_ms": 500.2, "request_id": "abc12345", ...}
```

### Question 5

**Scenario:** A script needs to monitor the health of multiple services by calling their health endpoints. It should log results and send an alert (simulated) if any service is unhealthy for 3 consecutive checks.

**Question:** Implement the health checker with logging and alerting.

**Answer:**

```python
import logging
import time
import requests
from collections import defaultdict
from datetime import datetime

# Configure logging
logging.basicConfig(
    level=logging.INFO,
    format='%(asctime)s - %(levelname)s - %(message)s'
)
logger = logging.getLogger(__name__)

class HealthChecker:
    def __init__(self, services, unhealthy_threshold=3, check_interval=30):
        self.services = services
        self.unhealthy_threshold = unhealthy_threshold
        self.check_interval = check_interval
        self.consecutive_failures = defaultdict(int)
        self.last_status = {}
    
    def check_service(self, name, url, timeout=10):
        """Check a single service's health endpoint"""
        try:
            start = time.time()
            response = requests.get(url, timeout=timeout)
            duration = (time.time() - start) * 1000
            
            if response.status_code == 200:
                return {
                    'status': 'healthy',
                    'status_code': response.status_code,
                    'response_time_ms': duration
                }
            else:
                return {
                    'status': 'unhealthy',
                    'status_code': response.status_code,
                    'response_time_ms': duration,
                    'error': f"HTTP {response.status_code}"
                }
                
        except requests.exceptions.Timeout:
            return {
                'status': 'unhealthy',
                'error': f"Timeout after {timeout}s",
                'response_time_ms': timeout * 1000
            }
        except requests.exceptions.ConnectionError:
            return {
                'status': 'unhealthy',
                'error': "Connection failed"
            }
        except Exception as e:
            return {
                'status': 'unhealthy',
                'error': str(e)
            }
    
    def send_alert(self, service_name, status, consecutive_failures):
        """Send alert (simulated)"""
        alert_msg = (
            f"🚨 ALERT: Service '{service_name}' is {status['status']}\n"
            f"   Error: {status.get('error', 'N/A')}\n"
            f"   Consecutive failures: {consecutive_failures}\n"
            f"   Time: {datetime.now().isoformat()}"
        )
        logger.warning(alert_msg)
        # In production, send to Slack, PagerDuty, email, etc.
        # send_slack_webhook(alert_msg)
    
    def run_once(self):
        """Run one health check cycle"""
        logger.info("Running health check cycle")
        
        for service in self.services:
            name = service['name']
            url = service['url']
            
            result = self.check_service(name, url)
            is_healthy = result['status'] == 'healthy'
            
            # Track consecutive failures
            if is_healthy:
                if self.consecutive_failures[name] >= self.unhealthy_threshold:
                    logger.info(f"Service '{name}' recovered")
                self.consecutive_failures[name] = 0
                self.last_status[name] = 'healthy'
                logger.info(f"✓ {name}: OK ({result['response_time_ms']:.0f}ms)")
            else:
                self.consecutive_failures[name] += 1
                self.last_status[name] = 'unhealthy'
                logger.warning(f"✗ {name}: FAILED - {result.get('error', 'Unknown error')}")
                
                # Alert if threshold reached
                if self.consecutive_failures[name] == self.unhealthy_threshold:
                    self.send_alert(name, result, self.consecutive_failures[name])
    
    def run_forever(self):
        """Run health checks continuously"""
        logger.info(f"Starting health checker (interval={self.check_interval}s, threshold={self.unhealthy_threshold})")
        
        while True:
            try:
                self.run_once()
            except KeyboardInterrupt:
                logger.info("Shutting down health checker")
                break
            except Exception as e:
                logger.exception(f"Health checker error: {e}")
            
            time.sleep(self.check_interval)

# Configuration
services = [
    {'name': 'web-api', 'url': 'https://api.example.com/health'},
    {'name': 'database', 'url': 'https://db.example.com/health'},
    {'name': 'cache', 'url': 'https://cache.example.com/health'}
]

# Run health checker
checker = HealthChecker(services, unhealthy_threshold=3, check_interval=30)
checker.run_forever()
```
