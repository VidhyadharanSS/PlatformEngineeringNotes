## Cheatsheet: Automation and System Integration

### subprocess.run() – Run Shell Commands

```python
import subprocess

# Basic run
result = subprocess.run(['ls', '-la'])

# Capture output
result = subprocess.run(['ls', '-la'], capture_output=True, text=True)
print(result.stdout)

# Check success (raises exception on failure)
subprocess.run(['ls', '/nonexistent'], check=True)

# With timeout
result = subprocess.run(['sleep', '10'], timeout=5)

# With environment
env = os.environ.copy()
env['DEBUG'] = 'true'
subprocess.run(['python', 'script.py'], env=env)
```

### subprocess.run() Parameters

| Parameter | Purpose | Default |
|-----------|---------|---------|
| `capture_output` | Capture stdout/stderr | `False` |
| `text` | Return strings (not bytes) | `False` |
| `check` | Raise on failure | `False` |
| `timeout` | Timeout in seconds | `None` |
| `shell` | Use shell (caution!) | `False` |

### pathlib – Modern Path Handling

```python
from pathlib import Path

# Create paths
current = Path.cwd()
home = Path.home()
path = Path('/etc/nginx/nginx.conf')
joined = Path('/var') / 'log' / 'nginx' / 'access.log'

# Check properties
path.exists()
path.is_file()
path.is_dir()

# Path components
path.name          # nginx.conf
path.stem          # nginx
path.suffix        # .conf
path.parent        # /etc/nginx

# Read/write files
content = Path('file.txt').read_text()
Path('output.txt').write_text('Hello')

# Directory operations
for file in Path('/var/log').glob('*.log'):
    print(file)

for py in Path('/home').rglob('*.py'):
    print(py)

# Create directories
Path('parent/child').mkdir(parents=True, exist_ok=True)
```

### pathlib vs os.path

| Operation | os.path | pathlib |
|-----------|---------|---------|
| Join paths | `os.path.join(a, b)` | `Path(a) / b` |
| Get filename | `os.path.basename(p)` | `Path(p).name` |
| Get directory | `os.path.dirname(p)` | `Path(p).parent` |
| Split extension | `os.path.splitext(p)` | `Path(p).stem`, `.suffix` |
| Check exists | `os.path.exists(p)` | `Path(p).exists()` |
| List directory | `os.listdir(d)` | `Path(d).iterdir()` |

| Feature | subprocess.run() | os.system() |
|---------|------------------|-------------|

### Question 1

**Scenario:** A Python script needs to run a long-running `kubectl rollout status` command that may take several minutes. If it takes longer than 5 minutes, it should time out and report failure.

**Question:** Write a function that implements this with proper timeout handling.

**Answer:**

```python
import subprocess
import sys

def rollout_status(deployment, namespace='default', timeout=300):
    """
    Check rollout status with timeout
    
    Args:
        deployment: Deployment name
        namespace: Kubernetes namespace
        timeout: Timeout in seconds (default 300 = 5 minutes)
    
    Returns:
        bool: True if rollout succeeded, False otherwise
    """
    cmd = ['kubectl', 'rollout', 'status', 'deployment', deployment, '-n', namespace]
    
    try:
        result = subprocess.run(
            cmd,
            capture_output=True,
            text=True,
            timeout=timeout
        )
        
        if result.returncode == 0:
            print(f"✅ Rollout successful: {deployment}")
            print(result.stdout)
            return True
        else:
            print(f"❌ Rollout failed: {deployment}")
            print(result.stderr, file=sys.stderr)
            return False
            
    except subprocess.TimeoutExpired:
        print(f"⏰ Rollout timed out after {timeout} seconds: {deployment}")
        return False
    except FileNotFoundError:
        print("❌ kubectl command not found. Is kubectl installed?", file=sys.stderr)
        return False

# Usage
if rollout_status('myapp', 'production', timeout=300):
    print("Deployment ready!")
else:
    sys.exit(1)
```

### Question 2

**Scenario:** A script needs to accept a configuration file path from the user. If not provided, it should look for `config.yaml` in the current directory, then in `~/.myapp/config.yaml`. If still not found, use default configuration.

**Question:** Write a function that resolves the config path using `pathlib`.

**Answer:**

```python
from pathlib import Path
import os

def find_config_file(custom_path=None):
    """
    Find configuration file in order:
    1. Custom path (if provided)
    2. ./config.yaml
    3. ~/.myapp/config.yaml
    4. Default config (return None)
    """
    if custom_path:
        path = Path(custom_path)
        if path.exists() and path.is_file():
            return path
    
    # Check current directory
    current = Path.cwd() / 'config.yaml'
    if current.exists():
        return current
    
    # Check user config directory
    user_config = Path.home() / '.myapp' / 'config.yaml'
    if user_config.exists():
        return user_config
    
    return None

def load_config(config_path=None):
    """Load configuration from YAML file"""
    import yaml
    
    path = find_config_file(config_path)
    
    if path:
        print(f"Loading config from: {path}")
        with open(path, 'r') as f:
            return yaml.safe_load(f)
    else:
        print("No config file found, using defaults")
        return {
            'server': {'host': 'localhost', 'port': 8080},
            'database': {'host': 'localhost', 'port': 5432}
        }

# Usage
config = load_config()
print(f"Server: {config['server']['host']}:{config['server']['port']}")
```

### Question 3

**Scenario:** A Python script needs to be used both as a CLI tool and imported as a module. It should parse arguments when run directly, but not interfere with imports.

**Question:** Write the pattern for this, including argument parsing for multiple commands.

**Answer:**

```python
#!/usr/bin/env python3
"""
backup_tool.py - Backup and restore utility
"""

import argparse
import shutil
import sys
from pathlib import Path

class BackupTool:
    def __init__(self, verbose=False):
        self.verbose = verbose
    
    def backup(self, source, destination):
        """Backup files from source to destination"""
        src = Path(source)
        dst = Path(destination)
        
        if not src.exists():
            raise FileNotFoundError(f"Source not found: {source}")
        
        dst.mkdir(parents=True, exist_ok=True)
        
        copied = 0
        for file in src.rglob('*'):
            if file.is_file():
                rel_path = file.relative_to(src)
                dest_file = dst / rel_path
                dest_file.parent.mkdir(parents=True, exist_ok=True)
                shutil.copy2(file, dest_file)
                copied += 1
                if self.verbose:
                    print(f"  Copied: {file} -> {dest_file}")
        
        print(f"✅ Backup complete: {copied} files copied")
        return copied
    
    def restore(self, backup_dir, destination):
        """Restore files from backup to destination"""
        backup = Path(backup_dir)
        dest = Path(destination)
        
        if not backup.exists():
            raise FileNotFoundError(f"Backup not found: {backup_dir}")
        
        dest.mkdir(parents=True, exist_ok=True)
        
        restored = 0
        for file in backup.rglob('*'):
            if file.is_file():
                rel_path = file.relative_to(backup)
                dest_file = dest / rel_path
                dest_file.parent.mkdir(parents=True, exist_ok=True)
                shutil.copy2(file, dest_file)
                restored += 1
                if self.verbose:
                    print(f"  Restored: {file} -> {dest_file}")
        
        print(f"✅ Restore complete: {restored} files restored")
        return restored

def parse_args():
    parser = argparse.ArgumentParser(description='Backup and restore tool')
    parser.add_argument('-v', '--verbose', action='store_true', help='Verbose output')
    
    subparsers = parser.add_subparsers(dest='command', help='Commands')
    
    # Backup command
    backup_parser = subparsers.add_parser('backup', help='Backup files')
    backup_parser.add_argument('source', help='Source directory')
    backup_parser.add_argument('destination', help='Destination directory')
    
    # Restore command
    restore_parser = subparsers.add_parser('restore', help='Restore from backup')
    restore_parser.add_argument('backup', help='Backup directory')
    restore_parser.add_argument('destination', help='Destination directory')
    
    return parser.parse_args()

def main():
    args = parse_args()
    
    if not args.command:
        print("Error: No command specified", file=sys.stderr)
        sys.exit(1)
    
    tool = BackupTool(verbose=args.verbose)
    
    try:
        if args.command == 'backup':
            tool.backup(args.source, args.destination)
        elif args.command == 'restore':
            tool.restore(args.backup, args.destination)
    except Exception as e:
        print(f"Error: {e}", file=sys.stderr)
        sys.exit(1)

if __name__ == '__main__':
    main()
```

### Question 4

**Scenario:** A script needs to find all YAML files in a directory tree and validate them with `kubectl apply --dry-run=client`. It should collect all errors and report at the end.

**Question:** Write a function that does this, using `subprocess` to run kubectl.

**Answer:**

```python
import subprocess
from pathlib import Path

def validate_kubernetes_manifests(directory):
    """Find and validate all YAML files in directory tree"""
    manifests = []
    errors = []
    
    # Find all .yaml and .yml files
    for yaml_file in Path(directory).rglob('*.yaml'):
        manifests.append(yaml_file)
    for yaml_file in Path(directory).rglob('*.yml'):
        manifests.append(yaml_file)
    
    print(f"Found {len(manifests)} manifest files")
    
    for manifest in manifests:
        print(f"Validating: {manifest}")
        
        # Run kubectl dry-run
        cmd = ['kubectl', 'apply', '--dry-run=client', '-f', str(manifest)]
        
        try:
            result = subprocess.run(
                cmd,
                capture_output=True,
                text=True,
                timeout=30
            )
            
            if result.returncode == 0:
                print(f"  ✅ Valid")
            else:
                error_msg = f"  ❌ Invalid: {manifest}\n{result.stderr}"
                errors.append(error_msg)
                print(error_msg)
                
        except subprocess.TimeoutExpired:
            error_msg = f"  ⏰ Timeout: {manifest}"
            errors.append(error_msg)
            print(error_msg)
        except FileNotFoundError:
            print("Error: kubectl command not found")
            return False
    
    # Report summary
    print(f"\n{'='*50}")
    print(f"Summary: {len(manifests)} files, {len(errors)} errors")
    
    if errors:
        print("\nErrors:")
        for error in errors:
            print(error)
        return False
    
    return True

# Usage
if __name__ == '__main__':
    import sys
    directory = sys.argv[1] if len(sys.argv) > 1 else '.'
    success = validate_kubernetes_manifests(directory)
    sys.exit(0 if success else 1)
```

### Question 5

**Scenario:** A script needs to accept a `--env` argument that can be `dev`, `staging`, or `prod`. Based on this, it should load different configuration files and set different environment variables.

**Question:** Write a CLI tool that implements this using `argparse` with choices.

**Answer:**

```python
#!/usr/bin/env python3
"""
deploy.py - Environment-based deployment tool
"""

import argparse
import os
import subprocess
import sys
from pathlib import Path

class Config:
    def __init__(self, environment):
        self.environment = environment
        self.load_config()
    
    def load_config(self):
        """Load configuration based on environment"""
        if self.environment == 'dev':
            self.namespace = 'development'
            self.replicas = 1
            self.log_level = 'DEBUG'
            self.config_file = 'config/dev.yaml'
        elif self.environment == 'staging':
            self.namespace = 'staging'
            self.replicas = 3
            self.log_level = 'INFO'
            self.config_file = 'config/staging.yaml'
        elif self.environment == 'prod':
            self.namespace = 'production'
            self.replicas = 10
            self.log_level = 'WARNING'
            self.config_file = 'config/prod.yaml'
        else:
            raise ValueError(f"Unknown environment: {self.environment}")
    
    def set_environment_variables(self):
        """Set environment variables for deployment"""
        os.environ['ENVIRONMENT'] = self.environment
        os.environ['NAMESPACE'] = self.namespace
        os.environ['LOG_LEVEL'] = self.log_level
        os.environ['REPLICAS'] = str(self.replicas)

def deploy(config):
    """Deploy to Kubernetes"""
    print(f"🚀 Deploying to {config.environment} environment")
    print(f"   Namespace: {config.namespace}")
    print(f"   Replicas: {config.replicas}")
    print(f"   Log level: {config.log_level}")
    
    # Set environment variables for subprocesses
    config.set_environment_variables()
    
    # Apply ConfigMap
    if Path(config.config_file).exists():
        cmd = ['kubectl', 'apply', '-f', config.config_file, '-n', config.namespace]
        result = subprocess.run(cmd, capture_output=True, text=True)
        if result.returncode != 0:
            print(f"Error applying config: {result.stderr}")
            return False
    
    # Apply deployment with replica count
    cmd = ['kubectl', 'scale', 'deployment', 'myapp', f'--replicas={config.replicas}', '-n', config.namespace]
    subprocess.run(cmd)
    
    # Rollout status
    cmd = ['kubectl', 'rollout', 'status', 'deployment', 'myapp', '-n', config.namespace]
    result = subprocess.run(cmd, capture_output=True, text=True)
    
    if result.returncode == 0:
        print("✅ Deployment successful!")
        return True
    else:
        print(f"❌ Deployment failed: {result.stderr}")
        return False

def main():
    parser = argparse.ArgumentParser(description='Deploy application to different environments')
    parser.add_argument(
        '--env',
        choices=['dev', 'staging', 'prod'],
        required=True,
        help='Target environment'
    )
    parser.add_argument('--dry-run', action='store_true', help='Show what would be done')
    
    args = parser.parse_args()
    
    config = Config(args.env)
    
    if args.dry-run:
        print(f"Would deploy to {args.env} environment")
        print(f"  Namespace: {config.namespace}")
        print(f"  Replicas: {config.replicas}")
        print(f"  Config file: {config.config_file}")
        sys.exit(0)
    
    success = deploy(config)
    sys.exit(0 if success else 1)

if __name__ == '__main__':
    main()
```
