### Mocking with `unittest.mock`

```python
from unittest.mock import Mock, patch

# Patch decorator
@patch('module.function')
def test_something(mock_func):
    mock_func.return_value = "mocked"

# Patch context manager
with patch('module.function') as mock_func:
    mock_func.return_value = "mocked"

# Mock object
mock = Mock()
mock.method.return_value = "value"
mock.method.assert_called_once()
```

### Retry Decorator Pattern

```python
def retry(max_attempts=3, delay=1, backoff=2):
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            for attempt in range(1, max_attempts + 1):
                try:
                    return func(*args, **kwargs)
                except Exception as e:
                    if attempt == max_attempts:
                        raise
                    time.sleep(delay * (backoff ** (attempt - 1)))
            return None
        return wrapper
    return decorator
```

|---------|--------|----------|
| Syntax | Simple assertions | `self.assertEqual()` |

### Retry Strategies

| Strategy | When to Use | Example |
|----------|-------------|---------|
| Fixed delay | Simple, fast operations | `time.sleep(1)` |
| Exponential backoff | Network calls | `delay * (2 ** attempt)` |
| With jitter | Many clients | `delay + random.uniform(0, delay*0.1)` |
| Circuit breaker | Cascading failures | Stop after N failures |

### Question 1: File Processing Script (15 minutes)

**Scenario:** Write a Python script that processes a log file and counts errors by severity (ERROR, WARNING, INFO). The script should:
- Accept a file path as a command-line argument
- Handle file not found errors gracefully
- Use proper logging
- Return exit code 0 on success, 1 on failure

**Answer:**

```python
#!/usr/bin/env python3
"""
log_analyzer.py - Analyze log files and count messages by severity
"""

import argparse
import logging
import sys
from collections import defaultdict
from pathlib import Path

def setup_logging(verbose=False):
    level = logging.DEBUG if verbose else logging.INFO
    logging.basicConfig(
        level=level,
        format='%(asctime)s - %(levelname)s - %(message)s'
    )

def parse_args():
    parser = argparse.ArgumentParser(description='Analyze log files')
    parser.add_argument('logfile', help='Path to log file')
    parser.add_argument('-v', '--verbose', action='store_true', help='Verbose output')
    return parser.parse_args()

def analyze_log(filepath):
    """Count log messages by severity"""
    counts = defaultdict(int)
    path = Path(filepath)
    
    if not path.exists():
        raise FileNotFoundError(f"Log file not found: {filepath}")
    
    with open(filepath, 'r') as f:
        for line in f:
            for severity in ['ERROR', 'WARNING', 'INFO', 'DEBUG']:
                if severity in line:
                    counts[severity] += 1
                    break
    
    return dict(counts)

def main():
    args = parse_args()
    setup_logging(args.verbose)
    logger = logging.getLogger(__name__)
    
    try:
        counts = analyze_log(args.logfile)
        
        print("Log Analysis Results:")
        print("-" * 30)
        for severity in ['ERROR', 'WARNING', 'INFO', 'DEBUG']:
            count = counts.get(severity, 0)
            print(f"{severity:8}: {count}")
        
        logger.info(f"Analysis complete: {sum(counts.values())} total messages")
        return 0
        
    except FileNotFoundError as e:
        logger.error(str(e))
        return 1
    except Exception as e:
        logger.exception(f"Unexpected error: {e}")
        return 1

if __name__ == "__main__":
    sys.exit(main())
```

### Question 2: API Client with Retry (15 minutes)

**Scenario:** Write an API client that fetches data from `https://api.example.com/users`. The client should:
- Retry up to 3 times on failure
- Use exponential backoff (1s, 2s, 4s)
- Log each retry attempt
- Timeout after 10 seconds
- Parse JSON response

**Answer:**

```python
import time
import logging
import requests
from functools import wraps

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

def retry(max_attempts=3, backoff=2, delay=1):
    """Retry decorator with exponential backoff"""
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            current_delay = delay
            for attempt in range(1, max_attempts + 1):
                try:
                    return func(*args, **kwargs)
                except Exception as e:
                    if attempt == max_attempts:
                        logger.error(f"Failed after {max_attempts} attempts")
                        raise
                    
                    logger.warning(
                        f"Attempt {attempt}/{max_attempts} failed: {e}. "
                        f"Retrying in {current_delay}s..."
                    )
                    time.sleep(current_delay)
                    current_delay *= backoff
            return None
        return wrapper
    return decorator

class APIClient:
    def __init__(self, base_url):
        self.base_url = base_url.rstrip('/')
    
    @retry(max_attempts=3, backoff=2, delay=1)
    def get_users(self):
        """Fetch users from API with retry"""
        url = f"{self.base_url}/users"
        
        response = requests.get(url, timeout=10)
        response.raise_for_status()
        
        return response.json()

# Usage
client = APIClient('https://api.example.com')
try:
    users = client.get_users()
    print(f"Fetched {len(users)} users")
except Exception as e:
    print(f"Failed to fetch users: {e}")
```

### Question 3: Kubernetes Deployment Script (20 minutes)

**Scenario:** Write a Python script that deploys a Kubernetes deployment using `kubectl`. The script should:
- Accept namespace, deployment name, and image tag as arguments
- Run `kubectl set image` and `kubectl rollout status`
- Handle subprocess errors
- Log all actions
- Have a `--dry-run` option

**Answer:**

```python
#!/usr/bin/env python3
"""
k8s_deploy.py - Deploy Kubernetes deployment with new image
"""

import argparse
import logging
import subprocess
import sys

def setup_logging(verbose=False):
    level = logging.DEBUG if verbose else logging.INFO
    logging.basicConfig(
        level=level,
        format='%(asctime)s - %(levelname)s - %(message)s'
    )

def parse_args():
    parser = argparse.ArgumentParser(description='Deploy Kubernetes deployment')
    parser.add_argument('-n', '--namespace', default='default', help='Kubernetes namespace')
    parser.add_argument('-d', '--deployment', required=True, help='Deployment name')
    parser.add_argument('-i', '--image', required=True, help='New image tag')
    parser.add_argument('-v', '--verbose', action='store_true', help='Verbose output')
    parser.add_argument('--dry-run', action='store_true', help='Dry run (no changes)')
    return parser.parse_args()

def run_kubectl(args, dry_run=False, capture=True):
    """Run kubectl command"""
    cmd = ['kubectl']
    
    if args.namespace:
        cmd.extend(['-n', args.namespace])
    
    cmd.extend(args.cmd_args)
    
    logging.info(f"Running: {' '.join(cmd)}")
    
    if dry_run:
        logging.info("[DRY RUN] Would execute command")
        return True, "dry run", ""
    
    result = subprocess.run(
        cmd,
        capture_output=capture,
        text=True
    )
    
    if result.returncode != 0:
        logging.error(f"kubectl failed: {result.stderr}")
        return False, result.stdout, result.stderr
    
    return True, result.stdout, result.stderr

def deploy(namespace, deployment, image, dry_run=False):
    """Deploy new image to Kubernetes"""
    logging.info(f"Deploying {deployment} with image {image} in namespace {namespace}")
    
    # Step 1: Set image
    success, stdout, stderr = run_kubectl(
        {'namespace': namespace, 'cmd_args': ['set', 'image', f'deployment/{deployment}', f'myapp={image}']},
        dry_run=dry_run
    )
    if not success:
        return False
    
    # Step 2: Wait for rollout
    success, stdout, stderr = run_kubectl(
        {'namespace': namespace, 'cmd_args': ['rollout', 'status', f'deployment/{deployment}', '--timeout=300s']},
        dry_run=dry_run
    )
    if not success:
        return False
    
    logging.info("Deployment successful")
    return True

def main():
    args = parse_args()
    setup_logging(args.verbose)
    
    if args.dry_run:
        logging.info("DRY RUN MODE - No changes will be made")
    
    success = deploy(args.namespace, args.deployment, args.image, args.dry_run)
    
    return 0 if success else 1

if __name__ == "__main__":
    sys.exit(main())
```

### Question 4: Configuration Loader (15 minutes)

**Scenario:** Write a configuration loader that:
- Reads from YAML file if provided
- Overrides with environment variables (prefix `MYAPP_`)
- Uses defaults for missing values
- Supports nested keys (e.g., `server.port`)

**Answer:**

```python
import os
import yaml
from pathlib import Path

class Config:
    DEFAULTS = {
        'server': {
            'host': '0.0.0.0',
            'port': 8080,
            'workers': 4
        },
        'database': {
            'host': 'localhost',
            'port': 5432,
            'name': 'myapp'
        },
        'logging': {
            'level': 'INFO'
        }
    }
    
    def __init__(self, config_path=None):
        self._config = self.DEFAULTS.copy()
        
        if config_path:
            self._load_from_file(config_path)
        
        self._load_from_env()
    
    def _load_from_file(self, path):
        path = Path(path)
        if not path.exists():
            raise FileNotFoundError(f"Config file not found: {path}")
        
        with open(path, 'r') as f:
            file_config = yaml.safe_load(f)
            self._deep_update(self._config, file_config)
    
    def _load_from_env(self):
        prefix = 'MYAPP_'
        for key, value in os.environ.items():
            if key.startswith(prefix):
                parts = key[len(prefix):].lower().split('__')
                self._set_nested(parts, value)
    
    def _set_nested(self, keys, value):
        config = self._config
        for key in keys[:-1]:
            config = config.setdefault(key, {})
        
        # Convert value type
        if value.lower() == 'true':
            value = True
        elif value.lower() == 'false':
            value = False
        elif value.isdigit():
            value = int(value)
        
        config[keys[-1]] = value
    
    def _deep_update(self, base, update):
        for key, value in update.items():
            if isinstance(value, dict) and key in base:
                self._deep_update(base[key], value)
            else:
                base[key] = value
    
    def get(self, path, default=None):
        keys = path.split('.')
        value = self._config
        for key in keys:
            if isinstance(value, dict):
                value = value.get(key)
                if value is None:
                    return default
            else:
                return default
        return value

# Usage
config = Config('/etc/myapp/config.yaml')
print(f"Server: {config.get('server.host')}:{config.get('server.port')}")
print(f"Database: {config.get('database.host')}:{config.get('database.port')}")
```

### Question 5: Graceful Shutdown (10 minutes)

**Scenario:** Write a script that processes a queue of items. It should:
- Handle SIGTERM and SIGINT signals
- Complete current item before shutting down
- Log shutdown status
- Clean up resources

**Answer:**

```python
#!/usr/bin/env python3
"""
queue_processor.py - Process items with graceful shutdown
"""

import signal
import time
import logging
import sys

logging.basicConfig(
    level=logging.INFO,
    format='%(asctime)s - %(levelname)s - %(message)s'
)
logger = logging.getLogger(__name__)

class GracefulShutdown:
    def __init__(self):
        self.shutdown_requested = False
        self.current_item = None
    
    def setup(self):
        signal.signal(signal.SIGTERM, self._handler)
        signal.signal(signal.SIGINT, self._handler)
        logger.info("Signal handlers registered")
    
    def _handler(self, signum, frame):
        logger.info(f"Received signal {signum}, initiating graceful shutdown...")
        self.shutdown_requested = True
    
    def should_shutdown(self):
        return self.shutdown_requested

def process_item(item):
    """Process a single item"""
    logger.info(f"Processing item: {item}")
    time.sleep(2)  # Simulate work
    logger.info(f"Completed item: {item}")

def main():
    graceful = GracefulShutdown()
    graceful.setup()
    
    items = list(range(1, 21))
    logger.info(f"Starting queue processor with {len(items)} items")
    
    try:
        for item in items:
            if graceful.should_shutdown():
                logger.warning(f"Shutdown requested. {len(items) - items.index(item)} items remaining")
                break
            
            graceful.current_item = item
            process_item(item)
            graceful.current_item = None
        
        logger.info("Queue processor finished")
        return 0
        
    except KeyboardInterrupt:
        logger.info("Interrupted by user")
        return 130
    except Exception as e:
        logger.exception(f"Unexpected error: {e}")
        return 1
    finally:
        if graceful.current_item:
            logger.info(f"Last completed item: {graceful.current_item}")
        logger.info("Cleanup complete")

if __name__ == "__main__":
    sys.exit(main())
```
