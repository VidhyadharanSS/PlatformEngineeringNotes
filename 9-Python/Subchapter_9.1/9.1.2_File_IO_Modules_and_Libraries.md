## Part 1: Reading and Writing Files

### Opening Files with `open()`

```python
# Open file for reading (default mode)
file = open('data.txt', 'r')
content = file.read()
file.close()  # Always close!

# Better: Use context manager (auto-closes)
with open('data.txt', 'r') as file:
    content = file.read()
    # No need to close - handled automatically
```

### File Modes

| Mode | Description |
|------|-------------|
| `'r'` | Read (default) |
| `'w'` | Write (overwrites) |
| `'a'` | Append (adds to end) |
| `'x'` | Exclusive creation (fails if exists) |
| `'b'` | Binary mode (e.g., `'rb'`, `'wb'`) |
| `'t'` | Text mode (default) |
| `'+'` | Read and write |

### Reading Files

```python
# Read entire file as string
with open('config.txt', 'r') as f:
    content = f.read()
    print(content)

# Read line by line (memory efficient for large files)
with open('large.log', 'r') as f:
    for line in f:
        print(line.strip())  # strip() removes newline

# Read all lines into list
with open('data.txt', 'r') as f:
    lines = f.readlines()  # Returns list of lines
    for line in lines:
        print(line)

# Read specific number of characters
with open('file.txt', 'r') as f:
    first_100 = f.read(100)
```

### Writing Files

```python
# Write string to file (overwrites)
with open('output.txt', 'w') as f:
    f.write("Hello, World!\n")
    f.write("Second line\n")

# Write list of lines
lines = ["Line 1\n", "Line 2\n", "Line 3\n"]
with open('output.txt', 'w') as f:
    f.writelines(lines)

# Append to file
with open('log.txt', 'a') as f:
    f.write("New log entry\n")

# Write binary data
with open('image.jpg', 'rb') as source:
    data = source.read()
with open('copy.jpg', 'wb') as dest:
    dest.write(data)
```

### Handling File Errors

```python
import os

try:
    with open('nonexistent.txt', 'r') as f:
        content = f.read()
except FileNotFoundError:
    print("File not found!")
except PermissionError:
    print("Permission denied!")
except IOError as e:
    print(f"I/O error: {e}")
```

## Part 2: The `os` Module – Operating System Interface

### File and Directory Operations

```python
import os

# Current working directory
cwd = os.getcwd()
print(f"Current directory: {cwd}")

# Change directory
os.chdir('/tmp')

# List directory contents
files = os.listdir('/var/log')
print(files[:5])  # First 5 files

# Create directory
os.mkdir('new_dir')
os.makedirs('parent/child/grandchild')  # Creates intermediate dirs

# Remove directory
os.rmdir('empty_dir')  # Only empty directories
import shutil
shutil.rmtree('non_empty_dir')  # Remove directory and contents

# Check if path exists
if os.path.exists('/etc/passwd'):
    print("passwd exists")

# Check if path is file or directory
print(os.path.isfile('/etc/passwd'))  # True
print(os.path.isdir('/etc'))          # True

# Get file size
size = os.path.getsize('/etc/passwd')
print(f"Size: {size} bytes")

# Get file modification time
import time
mtime = os.path.getmtime('/etc/passwd')
print(f"Modified: {time.ctime(mtime)}")
```

### Path Manipulation

```python
import os

# Join paths (cross-platform)
path = os.path.join('var', 'log', 'nginx', 'access.log')
print(path)  # var/log/nginx/access.log

# Split path
dir_name = os.path.dirname('/var/log/nginx/access.log')
base_name = os.path.basename('/var/log/nginx/access.log')
print(f"Directory: {dir_name}")  # /var/log/nginx
print(f"Filename: {base_name}")   # access.log

# Split extension
name, ext = os.path.splitext('archive.tar.gz')
print(f"Name: {name}, Extension: {ext}")  # archive.tar, .gz

# Get absolute path
abs_path = os.path.abspath('script.py')
print(abs_path)

# Expand user home
home = os.path.expanduser('~/.bashrc')
print(home)  # /home/user/.bashrc
```

### Environment Variables

```python
import os

# Get environment variable
db_host = os.environ.get('DB_HOST', 'localhost')
db_port = os.environ.get('DB_PORT', '5432')

# Set environment variable (affects current process only)
os.environ['APP_MODE'] = 'production'

# Check if variable exists
if 'PATH' in os.environ:
    print(os.environ['PATH'])

# List all environment variables
for key, value in os.environ.items():
    print(f"{key}={value}")
```

## Part 3: The `sys` Module – System-Specific Parameters

### Command-Line Arguments

```python
import sys

# Script name
print(f"Script: {sys.argv[0]}")

# Command-line arguments
print(f"Arguments: {sys.argv[1:]}")

# Example: python script.py --port 8080 --debug
if len(sys.argv) > 1:
    for arg in sys.argv[1:]:
        print(f"Arg: {arg}")
```

### Exit Codes

```python
import sys

def main():
    # Success
    sys.exit(0)
    
    # Or omit, script exits with 0 by default

if __name__ == "__main__":
    main()

# Failure
sys.exit(1)  # General error
sys.exit(2)  # Misuse of arguments
```

### Python Path and Version

```python
import sys

# Python version
print(f"Python {sys.version}")
print(f"Version info: {sys.version_info}")

# Module search path
for path in sys.path:
    print(path)

# Add custom module path
sys.path.append('/my/custom/path')
```

## Part 4: The `shutil` Module – High-Level File Operations

```python
import shutil

# Copy file
shutil.copy('source.txt', 'destination.txt')
shutil.copy2('source.txt', 'dest.txt')  # Preserves metadata

# Copy directory
shutil.copytree('src_dir', 'dst_dir')

# Move file/directory
shutil.move('old_location.txt', 'new_location.txt')

# Remove directory and contents
shutil.rmtree('temp_dir')

# Disk usage
usage = shutil.disk_usage('/')
print(f"Total: {usage.total / (1024**3):.1f} GB")
print(f"Used: {usage.used / (1024**3):.1f} GB")
print(f"Free: {usage.free / (1024**3):.1f} GB")

# Which command (like `which` in bash)
nginx_path = shutil.which('nginx')
print(f"nginx at: {nginx_path}")
```

## Part 5: Working with Config Files

### JSON

```python
import json

# Write JSON to file
data = {
    "name": "Alice",
    "age": 30,
    "email": "alice@example.com",
    "is_active": True
}

with open('user.json', 'w') as f:
    json.dump(data, f, indent=2)  # Pretty print

# Read JSON from file
with open('user.json', 'r') as f:
    loaded = json.load(f)
    print(loaded['name'])

# Parse JSON string
json_string = '{"name": "Bob", "age": 25}'
parsed = json.loads(json_string)
print(parsed['name'])

# Convert to JSON string
json_str = json.dumps(data, indent=2)
print(json_str)
```

## Part 6: The `datetime` Module

```python
from datetime import datetime, date, time, timedelta

# Current date and time
now = datetime.now()
print(f"Now: {now}")
print(f"Date: {now.date()}")
print(f"Time: {now.time()}")
print(f"Year: {now.year}, Month: {now.month}, Day: {now.day}")

# Create specific datetime
dt = datetime(2024, 1, 15, 10, 30, 45)

# Format datetime as string
formatted = now.strftime("%Y-%m-%d %H:%M:%S")
print(formatted)  # 2024-01-15 10:30:45

# Parse string to datetime
dt = datetime.strptime("2024-01-15 10:30:45", "%Y-%m-%d %H:%M:%S")

# Date arithmetic
tomorrow = now + timedelta(days=1)
last_week = now - timedelta(weeks=1)
next_hour = now + timedelta(hours=1)

# Compare dates
if now > last_week:
    print("Now is later")
```

## Part 7: The `re` Module – Regular Expressions

```python
import re

# Search for pattern
text = "The IP address is 192.168.1.100"
match = re.search(r'\d+\.\d+\.\d+\.\d+', text)
if match:
    print(f"Found IP: {match.group()}")

# Find all matches
log = "Error 404 at 10:00, Error 500 at 10:05"
errors = re.findall(r'Error \d+', log)
print(errors)  # ['Error 404', 'Error 500']

# Split string by pattern
csv_line = "apple,banana,orange"
items = re.split(r',', csv_line)
print(items)

# Replace pattern
text = "User alice logged in"
sanitized = re.sub(r'alice', 'REDACTED', text)
print(sanitized)

# Compile regex for reuse
ip_pattern = re.compile(r'\d+\.\d+\.\d+\.\d+')
matches = ip_pattern.findall("IPs: 192.168.1.1 and 10.0.0.1")
print(matches)  # ['192.168.1.1', '10.0.0.1']

# Groups
match = re.match(r'(\d{4})-(\d{2})-(\d{2})', '2024-01-15')
if match:
    year, month, day = match.groups()
    print(f"Year: {year}, Month: {month}, Day: {day}")
```

### Common Regex Patterns for Platform Engineers

| Pattern | Matches |
|---------|---------|
| `\d+\.\d+\.\d+\.\d+` | IP address |
| `\w+@\w+\.\w+` | Email (basic) |
| `\d{4}-\d{2}-\d{2}` | Date (YYYY-MM-DD) |
| `\d{2}:\d{2}:\d{2}` | Time (HH:MM:SS) |
| `https?://\S+` | URL |
| `[0-9a-f]{40}` | Git SHA |
| `\$\{(\w+)\}` | Variable substitution `${VAR}` |

## Part 8: Practical Example – Log Parser

```python
#!/usr/bin/env python3
import re
import sys
from datetime import datetime
from collections import defaultdict

def parse_nginx_log(log_file):
    """Parse Nginx access log and extract statistics"""
    pattern = r'(\d+\.\d+\.\d+\.\d+) - - \[(.*?)\] "(.*?)" (\d+) (\d+)'
    
    stats = {
        'total_requests': 0,
        'status_counts': defaultdict(int),
        'ip_counts': defaultdict(int),
        'error_requests': []
    }
    
    with open(log_file, 'r') as f:
        for line in f:
            match = re.match(pattern, line)
            if match:
                ip, timestamp, request, status, size = match.groups()
                
                stats['total_requests'] += 1
                stats['status_counts'][status] += 1
                stats['ip_counts'][ip] += 1
                
                if status.startswith('4') or status.startswith('5'):
                    stats['error_requests'].append({
                        'ip': ip,
                        'status': status,
                        'request': request,
                        'timestamp': timestamp
                    })
    
    return stats

def print_stats(stats):
    print(f"Total requests: {stats['total_requests']}")
    print("\nStatus codes:")
    for status, count in sorted(stats['status_counts'].items()):
        print(f"  {status}: {count}")
    
    print("\nTop 10 IPs:")
    top_ips = sorted(stats['ip_counts'].items(), key=lambda x: x[1], reverse=True)[:10]
    for ip, count in top_ips:
        print(f"  {ip}: {count}")
    
    print(f"\nError requests: {len(stats['error_requests'])}")
    for err in stats['error_requests'][:5]:
        print(f"  {err['timestamp']} - {err['ip']} - {err['status']}")

if __name__ == "__main__":
    if len(sys.argv) != 2:
        print(f"Usage: {sys.argv[0]} <log_file>")
        sys.exit(1)
    
    stats = parse_nginx_log(sys.argv[1])
    print_stats(stats)
```
